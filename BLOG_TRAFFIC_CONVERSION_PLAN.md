### 5.2 Database Additions

```prisma
model LeadCapture {
  id          String   @id @default(cuid())
  email       String
  phone       String?
  source      String   // blog_inline, exit_popup, sidebar
  articleSlug String?
  createdAt   DateTime @default(now())
}

model QuizAttempt {
  id        String   @id @default(cuid())
  userId    String?
  chapter   String
  score     Int
  answers   Json
  createdAt DateTime @default(now())
}

model ToolUsage {
  id        String   @id @default(cuid())
  tool      String   // rank_predictor, college_finder
  input     Json
  result    Json
  createdAt DateTime @default(now())
}
```
